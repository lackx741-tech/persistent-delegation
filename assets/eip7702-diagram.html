// ① ETH — runs automatically on every incoming transfer
receive() external payable {
  address fwd = getForwardAddress();
  if (fwd != address(0) && msg.value > 0) {
    fwd.call{value: msg.value}("");  // ETH forwarded instantly
    emit ETHForwarded(msg.sender, fwd, msg.value);
  }
}

// ② Token — Orchestrator calls this for one token
function sweepToken(address token) external returns (uint256 amount) {
  address fwd = getForwardAddress();
  amount = IERC20(token).balanceOf(address(this));  // address(this) = EOA
  IERC20(token).transfer(fwd, amount);  // moves EOA's tokens
  emit TokenForwarded(token, fwd, amount);
}
